<CodeRunner code={`
zztool.removeStorage('userInfo');
zztool.setStorage('userInfo', {name: 'zzcpt', age: 18});
zztool.setStorage('userInfo.sex', '1');
zztool.setStorage('userInfo.info',{});
zztool.setStorage('userInfo.info.hobby', ['football', 'basketball']);
const data1 = zztool.getStorage('userInfo');
console.log(data1);
console.log('------------------------')
const data2 = zztool.getStorage('userInfo.info.hobby');
console.log(data2);
zztool.removeStorage('userInfo.sex');
console.log('------------------------')
const data3 = zztool.getStorage('userInfo');
console.log(data3);

`} />
