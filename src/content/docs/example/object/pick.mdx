const data = zztool.pick(obj, "c")

console.log(data)
`} />

| --- | --- | --- | --- | --- |
| *data* | *数据* | *\** | *✅* | *-* |
| *index* | *索引* | *String* | *✅* | *-* |
{/* | *first* | *首字符* | *String* | *❌* | *空字符* | */}
{/* | *format* | *格式* | *String* | *❌* | *hh:mm* | */}
